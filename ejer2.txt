#include <iostream>
#include <vector>
#include <random>
#include <algorithm>

using std::vector; using std:: cin; using std::cout; using std::endl;

template<typename T>
vector<T> func(T n){
    std::random_device rd;
    std::default_random_engine re(rd());
    std::uniform_int_distribution<int> ud(1,50);
    vector<T>vec (n);
    for(auto& item: vec)
        item = ud(rd);
    sort(vec.begin(), vec.end());
    vec.erase(unique(vec.begin(), vec.end()),vec.end());
    return vec;
}

int main() {
    auto vect = func(500);
    for(auto it : vect)
        cout << it << " ";
    return 0;
}
